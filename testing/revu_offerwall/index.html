    action_id, 
    currency_amount, 
    response_code,
    response_body,
    processed_at,
    raw_params->>'secret' as signature_received
FROM postback_log 
WHERE user_id = '7736fa04-1267-40c5-9643-008630cf8de4'
ORDER BY processed_at DESC 
LIMIT 3;

-- 2. Check user balance (should increase by currency_amount)

FROM public.users 
WHERE id = '7736fa04-1267-40c5-9643-008630cf8de4';

-- 3. Check for fraud events (should be EMPTY if signature valid)

FROM fraud_events 
WHERE user_id = '7736fa04-1267-40c5-9643-008630cf8de4'
